questions = ['Where do Nirupam live?', 'Which operating system do Nirupam use?', 'Is Nirupam a student?']

answers = ['Hyderabad', 'Windows', 'Yes']

num_rights = 0

for i in range(len(questions)):

guess = input(questions[i])

if guess.lower() == answers[i].lower():

print('Correct')

num_rights = num_rights + 1

else:

print('Wrong,The answer is', answers[i])

print('You have', num_rights, 'out of', i, 'right')
